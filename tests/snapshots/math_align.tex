\section{Long Equation}
\begin{align}
P(A|B) &= \frac{P(B|A) P(A)}{P(B)} \\
       &= \frac{P(B|A) P(A)}{P(B|A) P(A) + P(B|\neg A) P(\neg A)}
\end{align}
